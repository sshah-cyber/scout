01 WA State Parcel API — all 39 counties
// Washington State GeoData Portal — statewide parcel layer
// https://geo.wa.gov/datasets/WaDNR::wa-state-parcels
// No API key required — public ArcGIS FeatureServer

const WA_PARCELS = 'https://services.arcgis.com/jsIt88o09Q0r1j8h/arcgis/rest/services/Washington_State_Parcels/FeatureServer/0';

async function waSearchByAddress(address, county = '') {
  let where = `UPPER(SITE_ADDRESS) LIKE UPPER('%${address}%')`;
  if (county) where += ` AND UPPER(COUNTY_NM) = UPPER('${county}')`;
  const p = new URLSearchParams({
    where, f: 'json',
    outFields: 'PARCEL_ID,SITE_ADDRESS,COUNTY_NM,LAND_SQFT,ZONING,USE_CODE,YEAR_BUILT,OWNER_NM',
    returnGeometry: false,
    resultRecordCount: 25
  });
  const r = await fetch(`${WA_PARCELS}/query?${p}`);
  const { features } = await r.json();
  return (features || []).map(f => ({ ...f.attributes, _src: 'WA State' }));
}

// King County — dedicated ArcGIS (richer data)
const KC = 'https://gismaps.kingcounty.gov/arcgis/rest/services/property/KingCountySitus/MapServer/0';

async function kcSearch(address) {
  const p = new URLSearchParams({
    where: `UPPER(ADDR_FULL) LIKE UPPER('%${address}%')`,
    outFields: 'PIN,ADDR_FULL,LOT_AREA,ZONE_CLASS,PRESENT_USE,YEAR_BUILT',
    returnGeometry: false, resultRecordCount: 20, f: 'json'
  });
  const { features } = await (await fetch(`${KC}/query?${p}`)).json();
  return (features || []).map(f => ({ ...f.attributes, _src: 'King County GIS', _county: 'King' }));
}

// Snohomish County
const SNO = 'https://gis.snoco.org/maps/rest/services/property/assessor/MapServer/0';

async function snoSearch(address) {
  const p = new URLSearchParams({
    where: `UPPER(SITE_ADDRESS) LIKE UPPER('%${address}%')`,
    outFields: 'PARCEL_NUM,SITE_ADDRESS,LAND_SQFT,ZONE_CODE,USE_CODE,YEAR_BUILT',
    returnGeometry: false, resultRecordCount: 20, f: 'json'
  });
  const { features } = await (await fetch(`${SNO}/query?${p}`)).json();
  return (features || []).map(f => ({
    PIN: f.attributes.PARCEL_NUM, ADDR_FULL: f.attributes.SITE_ADDRESS,
    LOT_AREA: f.attributes.LAND_SQFT, ZONE_CLASS: f.attributes.ZONE_CODE,
    PRESENT_USE: f.attributes.USE_CODE, YEAR_BUILT: f.attributes.YEAR_BUILT,
    _src: 'Snohomish GIS', _county: 'Snohomish'
  }));
}
02 HB 1337 ADU Eligibility Engine
// Statewide HB 1337 eligibility — applies to all WA counties
// Handles both urban (UGA) and rural parcels per HB 1337 + HB 1345

function assessAdu(attr) {
  const zone   = (attr.ZONE_CLASS || attr.ZONING || '').toUpperCase();
  const lot    = parseFloat(attr.LOT_AREA || attr.LAND_SQFT) || 0;
  const use    = (attr.PRESENT_USE || attr.USE_CODE || '').toUpperCase();
  const county = (attr._county || attr.COUNTY_NM || '').toUpperCase();

  // County-specific minimum lot sizes
  const minLotMap = {
    'KING': 3200, 'SNOHOMISH': 4800, 'PIERCE': 4000,
    'CLARK': 4000, 'SPOKANE': 3000, 'THURSTON': 3500,
    'WHATCOM': 4000, 'KITSAP': 4000
  };
  const minLot = minLotMap[county] || 4000; // HB 1337 default

  // SF zone detection — covers common WA zone codes
  const sfPatterns = [
    'SF', 'R-SF', 'RS', 'R1', 'R2', 'R4', 'R6', 'R8',
    'SFR', 'LDR', 'RES', 'RL', 'SR', 'RR', 'RE'
  ];
  const isSF  = sfPatterns.some(z => zone.includes(z)) || zone.match(/^R-?\d/);
  const sfUse = use.includes('SINGLE') || use.includes('SFR') ||
                use.includes('RES') || use === '';
  const lotOk = lot >= minLot;

  const checks = [], score = { v: 0 };
  const add = (pass, text, pts) => { checks.push({ pass, text }); if (pass) score.v += pts; };

  add(isSF,  isSF  ? `Zone ${zone || '—'} — ADUs permitted under HB 1337`
                   : `Zone ${zone || '—'} — verify ADU eligibility with local planning dept`, 35);
  add(lotOk, lotOk ? `Lot ${lot.toLocaleString()} sqft ≥ ${minLot.toLocaleString()} sqft local minimum`
                   : `Lot ${lot.toLocaleString()} sqft below ${minLot.toLocaleString()} sqft minimum — verify with city`, 30);
  add(sfUse, sfUse ? 'Single-family residential use confirmed'
                   : `Use type may limit ADU — confirm with local planning dept`, 10);
  add(true,  'No owner-occupancy required statewide (HB 1337)', 5);
  add(true,  'DADU condo sale permitted — city cannot prohibit (HB 1337 §4(1)(k))', 5);
  add(true,  'HOA restrictions unenforceable for HOAs formed after July 23, 2023', 3);
  add(true,  'WA capital gains tax exempt for real estate sales (WAC 458-20-185)', 2);

  if (lot < minLot + 1500 && lotOk) add(false, 'Lot is marginal — confirm rear yard depth before designing DADU', 0);

  const eligible = (!isSF || !lotOk) ? 'ineligible'
    : lot < minLot + 800 ? 'conditional' : 'eligible';
  const types = eligible === 'ineligible' ? [] : lot >= minLot + 800 ? ['DADU','AADU'] : ['AADU'];

  return { eligible, score: Math.min(100, Math.round(score.v)), types, checks };
}
